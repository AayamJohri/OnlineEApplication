public void updateProductQuantities() {
        for (int i = 0; i < productsOrdered.size(); i++) {
            Product product = productsOrdered.get(i);
            int purchasedQuantity = quantitiesOrdered.get(i);
            ProductCatalog productCatalog = new ProductCatalog();
            Product catalogProduct = productCatalog.getProductByID(product.getProductID());
            if (catalogProduct != null) {
                int availableQuantity = catalogProduct.getQuantityAvailable();
                if (purchasedQuantity <= availableQuantity) {
                    catalogProduct.updateQuantity(purchasedQuantity);
                } else {
                    System.out.println("Not enough quantity available for product: " + product.getProductName());
                }
            }
        }
   }
